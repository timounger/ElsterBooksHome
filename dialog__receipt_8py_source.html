300 def detect_data(self) -> None:
301 """!
302 @brief Detect data.
303 """
304 is_income = bool(self.receipt_type == EReceiptType.INCOME)
305 file_path = self.file_path
306 company = self.ui.tab_settings.company_data
307 company_default = company[COMPANY_DEFAULT_FIELD]
308 b_german = bool(company[COMPANY_ADDRESS_FIELD][ECompanyFields.COUNTRY] == "DE")
309 if file_path.lower().endswith(PDF_TYPE.lower()):
310 # check for auto import
311 if is_income:
312 b_custom_detect = False
313 else:
314 b_custom_detect = False
315 if not b_custom_detect:
316 if b_german and check_pre_tax(file_path):
317 data = import_pre_tax(file_path, is_income=is_income)
318 if data:
319 self.data = data
320 self.lbl_ai_assist_status.setText(f"{EReceiptGroup.UST_VA.value} erkannt!")
321 self.lbl_ai_assist_status.setStyleSheet("color: green;")
322 else:
323 self.lbl_ai_assist_status.setText(f"{EReceiptGroup.UST_VA.value} ist keine Ausgabe")
324 self.lbl_ai_assist_status.setStyleSheet("color: red;")
325 elif b_german and check_ust(file_path):
326 data = import_ust(file_path, is_income=is_income)
327 if data:
328 self.data = data
329 self.lbl_ai_assist_status.setText(f"{EReceiptGroup.UST.value} erkannt!")
330 self.lbl_ai_assist_status.setStyleSheet("color: green;")
331 else:
332 self.lbl_ai_assist_status.setText(f"{EReceiptGroup.UST.value} ist keine Ausgabe")
333 self.lbl_ai_assist_status.setStyleSheet("color: red;")
334 elif check_zugferd(file_path):
335 data = import_zugferd(file_path, is_income=is_income,
336 income_group=company_default[ECompanyFields.INCOME_GROUP],
337 expenditure_group=company_default[ECompanyFields.EXPENDITURE_GROUP])
338 if data:
339 self.data = data
340 self.lbl_ai_assist_status.setText("ZUGFeRD erkannt!")
341 self.lbl_ai_assist_status.setStyleSheet("color: green;")
342 else:
343 self.lbl_ai_assist_status.setText("ZUGFeRD konnte nicht automatisch importiert werden")
344 self.lbl_ai_assist_status.setStyleSheet("color: red;")
345 else:
346 if check_xinvoice(file_path):

348 income_group=company_default[ECompanyFields.INCOME_GROUP],
349 expenditure_group=company_default[ECompanyFields.EXPENDITURE_GROUP])

368 def update_pdf_page(self, direction: bool | None) -> None:
369 """!
370 @brief Update PDF page
371 @param direction : page direction (None: init first, False: prev, True: next)
372 """
373 total_pages = self.pdf_document.pageCount()
374 if total_pages > 1:
375 current_page = self.pdf_view.pageNavigator().currentPage() + 1
376 self.lbl_page_number.show()
377 if direction is None:
378 page_number = 1
379 b_light_theme = self.ui.model.c_monitor.is_light_theme()
380 self.btn_left.setIcon(QIcon(ICON_ARROW_LEFT_LIGHT if b_light_theme else ICON_ARROW_LEFT_DARK))
381 self.btn_right.setIcon(QIcon(ICON_ARROW_RIGHT_LIGHT if b_light_theme else ICON_ARROW_RIGHT_DARK))
382 self.btn_left.show()
383 self.btn_right.show()
384 self.btn_left.setEnabled(False)
385 self.btn_right.setEnabled(True)
386 else:
387 if direction:
388 page_number = current_page + 1
389 else:
390 page_number = current_page - 1
391 if page_number == 1:
392 self.btn_left.setEnabled(False)
393 self.btn_right.setEnabled(True)
394 elif page_number == total_pages:
395 self.btn_left.setEnabled(True)
396 self.btn_right.setEnabled(False)
397 self.pdf_view.pageNavigator().jump(page_number - 1, QPointF(0, 0))
398 self.lbl_page_number.setText(f"{page_number}/{total_pages}")

406 def doc_view_changed(self, doc_view: DocView) -> None:
407 """!
408 @brief Doc view changed. Set doc widget and update button color
409 @param doc_view : doc view index
410 """
411 if self.ui.model.c_monitor.is_light_theme():
412 active_fg = "white"
413 active_bg = "#007BFF"
414 inactive_fg = "#555555"
415 inactive_bg = "#e0e0e0"
416 else:
417 active_fg = "black"
418 active_bg = "#3399FF"
419 inactive_fg = "#bbbbbb"
420 inactive_bg = "#2e2e2e"
421 match doc_view:
422 case DocView.PDF:
423 self.doc_view.setCurrentIndex(DocView.PDF)
424 self.btn_view_pdf.setStyleSheet(f"color: {active_fg}; background-color: {active_bg};")
425 self.btn_view_xml.setStyleSheet(f"color: {inactive_fg}; background-color: {inactive_bg};")

431 self.btn_left.hide()
432 self.btn_right.hide()
433 self.lbl_page_number.hide()
434 case _:

445 def set_detected_invoice_data(self, invoice_data: InvoiceData) -> None:
446 """!
447 @brief Set detected invoice data. Do not add if user input text before
448 @param invoice_data : invoice data
449 """
450 if invoice_data is not None:
451 # invoice number
452 if invoice_data.invoice_number and not self.le_invoice_number.text():
453 self.le_invoice_number.setText(invoice_data.invoice_number)
454 # invoice data
455 current_date = QDate.currentDate()
456 if invoice_data.invoice_date and (self.de_invoice_date.date() == current_date):
457 invoice_date = QDate.fromString(invoice_data.invoice_date, DATE_FORMAT)
458 self.de_invoice_date.setDate(invoice_date)
459 # trade partner
460 if invoice_data.seller_name and not self.pte_trade_partner.toPlainText():
461 self.pte_trade_partner.setPlainText(invoice_data.seller_name)
462 # description
463 if invoice_data.description and not self.pte_description.toPlainText():
464 self.pte_description.setPlainText(invoice_data.description)
465 # gross
466 if invoice_data.gross_amount and not self.dsb_gross.value():
467 self.dsb_gross.setValue(invoice_data.gross_amount)
468 # net
469 if invoice_data.net_amount and not self.dsb_net.value():
470 self.dsb_net.setValue(invoice_data.net_amount)
471 self.lbl_ai_assist_status.setText("Erkennung abgeschlossen!")
472 self.lbl_ai_assist_status.setStyleSheet("color: green;")

474 def gross_changed(self) -> None:
475 """!
476 @brief Gross changed.
477 """
478 gross_price = self.dsb_gross.value()
479 if gross_price is not None:
480 self.b_gross_changed = True
481 if not self.b_net_changed:
482 if (self.receipt_type == EReceiptType.INCOME) and self.ui.tab_settings.company_data[COMPANY_BOOKING_FIELD][ECompanyFields.SMALL_BUSINESS_REGULATION]:
483 f_net = gross_price # income for small business regulation is net = gross
484 else:
485 f_net = round(gross_price / (1 + (self.default_tax_rate / 100)), 2)
486 self.dsb_net.setValue(f_net)
487 self.b_net_changed = False
488 else:
489 self.b_gross_changed = False
490 self.set_vat_rate()
491
492 def net_changed(self) -> None:
493 """!
494 @brief Net changed.
495 """
496 net_price = self.dsb_net.value()
497 if net_price is not None:
498 self.b_net_changed = True
499 if not self.b_gross_changed:
500 if (self.receipt_type == EReceiptType.INCOME) and self.ui.tab_settings.company_data[COMPANY_BOOKING_FIELD][ECompanyFields.SMALL_BUSINESS_REGULATION]:
501 f_gross = net_price # income for small business regulation is net = gross
502 else:
503 f_gross = round(net_price * (1 + (self.default_tax_rate / 100)), 2)
504 self.dsb_gross.setValue(f_gross)
505 self.b_gross_changed = False
506 else:
507 self.b_net_changed = False
508 self.set_vat_rate()
