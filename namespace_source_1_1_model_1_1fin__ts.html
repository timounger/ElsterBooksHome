128def create_transaction_xls_file(l_transactions: list[Transaction]) -> None:
129 """!
130 @brief Create transaction XLS Report
131 @param l_transactions : transactions
132 """
133 header = ["Datum", "Betrag", "Auftraggeber", "Verwendungszweck", "Buchungstext", "Referenz"]
134 if not os.path.exists(EXPORT_PATH):
135 os.makedirs(EXPORT_PATH)
136 xls_creator = XLSCreator()
137 worksheet = xls_creator.workbook.active
138 worksheet.title = "Transaktionen"
139 worksheet.column_dimensions["A"].width = 10 # Datum
140 worksheet.column_dimensions["B"].width = 15 # Betrag
141 worksheet.column_dimensions["C"].width = 30 # Auftraggeber
142 worksheet.column_dimensions["D"].width = 50 # Verwendungszweck
143 worksheet.column_dimensions["E"].width = 20 # Buchungstext
144 worksheet.column_dimensions["F"].width = 20 # Referenz
145 for i, title in enumerate(header):
146 xls_creator.set_cell(worksheet, 1, i + 1, title, bold=True)
147 i_transaction = 0
148 for transaction in reversed(l_transactions):
149 color = COLOR_RED_DARK if transaction.amount >= 0 else COLOR_RED
150 xls_creator.set_cell(worksheet, i_transaction + 2, 1, transaction.date.strftime(DATE_FORMAT_JSON), number_format=NUMBER_FORMAT_DATETIME)
151 xls_creator.set_cell(worksheet, i_transaction + 2, 2, transaction.amount, color=color, number_format=NUMBER_FORMAT_EUR)
152 xls_creator.set_cell(worksheet, i_transaction + 2, 3, str(transaction.applicant_name))
153 xls_creator.set_cell(worksheet, i_transaction + 2, 4, str(transaction.purpose))
154 xls_creator.set_cell(worksheet, i_transaction + 2, 5, str(transaction.text))
155 xls_creator.set_cell(worksheet, i_transaction + 2, 6, str(transaction.customer_reference))
156 i_transaction += 1
157 xls_creator.set_table(worksheet, max_col=len(header), max_row=i_transaction + 1, min_col=1, min_row=1)
158 xls_creator.save(TRANSACTIONS_XLS_FILE_PATH)

80def get_fints_institutes() -> list[FinTSInstitute]:
81 """!
82 @brief Get FinTS institute
83 @return FinTS institutes
84 """
85 l_institutes = []
86 with open(FINTS_INSTITUTE_FILE, mode="r", encoding="latin1", newline="") as csv_file:
87 reader = csv.reader(csv_file, delimiter=";", quotechar='"')
88 for i, row in enumerate(reader):
89 if i != 0:
90 blz = row[1]
91 bic = row[2]
92 instutute = row[3]
93 url = row[24]
94 if blz and instutute and url:
95 l_institutes.append(FinTSInstitute(blz, bic, instutute, url))
96 return l_institutes

99def read_data_from_transaction_file(transaction_file: str = TRANSACTIONS_CSV_FILE_PATH) -> list[list[str]]:
