73def add_document(path: str, add: bool, d_document_data: dict[EDocumentFields, str], document_id: Optional[str] = None,
74 appendix_file: Optional[str] = None, rename: bool = False) -> None:
75 """!
76 @brief Add or actualize document (data and appendix).
77 @param path : export to this path
78 @param add : GIT add status
79 @param d_document_data : document data to export
80 @param document_id : document ID
81 @param appendix_file : document file
82 @param rename : status if file name should renamed depend on actual data
83 """
84 s_id = set_general_json_data(d_document_data, DOCUMENT_TYPE, EDocumentFields.JSON_TYPE,
85 EDocumentFields.JSON_VERSION, JSON_VERSION_DOCUMENT,
86 EDocumentFields.ID, document_id)
87 date = d_document_data[EDocumentFields.DOCUMENT_DATE]
88 title = get_file_name([get_date_title(date), d_document_data[EDocumentFields.DESCRIPTION]], s_id)
89 if (appendix_file is not None) or rename:
90 if appendix_file is None:
91 if d_document_data[EDocumentFields.ATTACHMENT]:
92 _, file_extension = os.path.splitext(d_document_data[EDocumentFields.ATTACHMENT])
93 file_extension = file_extension.lower()
94 else:
95 file_extension = PDF_TYPE # only for compatible with old version without existing attachment
96 else:
97 _, file_extension = os.path.splitext(appendix_file)
98 file_extension = file_extension.lower()
99 d_document_data[EDocumentFields.ATTACHMENT] = f"{title}{file_extension}"
100 id_field = EDocumentFields.ID if (document_id is not None) else None
101 instance = fill_data(D_DOCUMENT_TEMPLATE, d_document_data)
102 add_json(add, instance, title, s_id, os.path.join(path, DOCUMENT_FOLDER), id_field=id_field, rename=rename)
103 if (appendix_file is not None) or rename:
104 add_appendix(title, s_id, os.path.join(path, DOCUMENT_FILE_PATH), add, appendix_file=appendix_file)
