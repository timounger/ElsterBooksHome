45def create_user_message(document_text: str) -> str:
46 """!
47 @brief Create user message for Ollama request
48 @param document_text : document text
49 @return user message text
50 """
51 document_text = document_text[:I_MAX_DOCUMENT_CHARS]
52
53 l_user_message = [
54 "You are a helpful assistant. Tell me about document data.",
55 "pattern code of invoice_date is: ^[0-9]{2}.[0-9]{2}.[0-9]{4}$",
56 "do not answer with pattern code string. Only with a valid date in this format",
57 "If not date detected return empty string",
58 "For description create maximal 3 words as summary of invoice goods or service",
59 "Invoice Text:",
60 document_text
61 ]
62
63 user_message = "\n".join(l_user_message)
64 return user_message

67def validate_answer(invoice_data: InvoiceData) -> InvoiceData:
68 """!
69 @brief Validate invoice data answer and fix if possible
70 @param invoice_data : invoice data
71 @return invoice data
72 """
73 if not is_date_format(invoice_data.invoice_date):
74 invoice_data.invoice_date = invoice_data.invoice_date.replace(" ", ".")
75 for i_month, s_month in enumerate(L_MONTH_NAMES, start=1):
76 invoice_data.invoice_date = invoice_data.invoice_date.replace(s_month, str(i_month).zfill(2))
77 if not is_date_format(invoice_data.invoice_date):
78 invoice_data.invoice_date = ""
79 return invoice_data

82def get_ai_document_text(file_path: str) -> str:
83 """!
84 @brief Get text from PDF for AI

89 if len(text) < I_MIN_OCR_TEXT:
