155def group_menu(parent: "MainWindow", l_actions: list[QAction], actual_value: Any, l_match_values: list[Any]) -> QActionGroup:
156 """!
157 @brief Set action group.
158 @param parent : parent window controller
159 @param l_actions: list with actions to add
160 @param actual_value : actual setting value
161 @param l_match_values : list with possible match values to check default setting
162 @return action group object
163 """
164 action_group = QActionGroup(parent)
165 for action in l_actions:
166 action_group.addAction(action)
167 for i, match_value in enumerate(l_match_values):
168 if match_value == actual_value:
169 l_actions[i].setChecked(True)
170 break
171 return action_group
