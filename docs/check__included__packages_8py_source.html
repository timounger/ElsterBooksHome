110def check_included_packages() -> list[str]:
111 """!
112 @brief Check included packages
113 @return status if included packages are okay
114 """
115 d_third_party = {}
116 d_buildin = {}
117 d_own_packs = {}
118
119 # Regular Expressions
120 regex_third_party = re.compile(r".venv\/lib\/site-packages.*.py", re.IGNORECASE)
121 regex_third_party_name = re.compile(r"(?<=site-packages\/).*", re.IGNORECASE)
122 regex_builtin = re.compile(r".venv\/lib\/.*.py", re.IGNORECASE)
123 regex_builtin_name = re.compile(r"(?<=lib\/).*", re.IGNORECASE)
124 regex_own_packs = re.compile(fr"{__title__}\/.*")
125
126 # read PyInstaller modulegraph cross reference HTML
127 with open(S_RELATIVE_PATH, mode="r", encoding="utf-8") as o_html_file:
128 soup = BeautifulSoup(o_html_file, "html.parser")
129
130 # parse HTML
131 l_nodes = soup.find_all("div", class_="node")
132 for o_node in l_nodes:
133 l_targets = o_node.find_all("a", target="code")
134 for o_target in l_targets:
135 s_package_path = o_target["href"]
136 if regex_third_party.search(s_package_path):
137 matches = regex_third_party_name.search(s_package_path)
138 if matches is not None:
139 d_third_party[matches.group().split("/", maxsplit=1)[0].replace(".py", "")] = ""
140 elif regex_builtin.search(s_package_path):
141 matches = regex_builtin_name.search(s_package_path)
142 if matches is not None:
143 d_buildin[matches.group().replace(".py", "")] = ""
144 elif regex_own_packs.search(s_package_path):
145 matches = regex_own_packs.search(s_package_path)
146 if matches is not None:
147 d_own_packs[matches.group()] = ""
148
149 # print included packages
150 log.debug("\nThird party packages:")
151 log.debug("\n".join(list(d_third_party.keys())))
152 log.debug("\nOwn packages:")
153 log.debug("\n".join(list(d_own_packs.keys())))
154 log.debug("\nPython buildin packages:")
155 log.debug("\n".join(list(d_buildin.keys())))
156
157 # check third party packages
158 l_not_allowed_packages = []
159 for s_package in list(d_third_party.keys()):
160 if s_package not in L_ALLOWED_THIRD_PARTY_PACKAGES:
161 l_not_allowed_packages.append(f"ERROR PyInstaller included an unknown package in the executable: \'{s_package}\'")
162 return l_not_allowed_packages
