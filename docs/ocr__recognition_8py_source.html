26def convert_pdf_to_images(file: str) -> list[PILImage]:
27 """!
28 @brief Convert PDF to images
29 @param file : file to convert
30 @return images for OCR recognition
31 """
32 pdf_document = fitz.open(file)
33 images = []
34
35 for page in pdf_document:
36 pix = page.get_pixmap(dpi=300)
37 image = Image.frombytes("RGB", (pix.width, pix.height), pix.samples)
38 images.append(image)
39
40 return images
