54def check_vat_format(vat_id: str, pattern: Optional[str] = None) -> bool:
55 """!
56 @brief Check for valid VAT format depend on country
57 @param vat_id : VAT id
58 @param pattern : check fix pattern else use VAT_PATTERNS
59 @return valid vat format
60 """
61 valid = False
62 if pattern:
63 if re.match(pattern, vat_id) is not None:
64 valid = True
65 else:
66 for country_code, actual_pattern in VAT_PATTERNS.items():
67 if vat_id.startswith(country_code):
68 if re.match(actual_pattern, vat_id) is not None:
69 valid = True
70 break
71 return valid

84 def validate_vat(self) -> dict[str, Any] | None:
85 """!
86 @brief Validate VAT
87 @return response status
88 """
89 vat_result = None
90 wsdl = 'http://ec.europa.eu/taxation_customs/vies/checkVatService.wsdl' # do not use https to prevent warning
91
92 country_code = self.vat_id[:2]
93 vat_number = self.vat_id[2:]
94
95 try:
96 client = Client(wsdl)
97 result = client.service.checkVat(
98 countryCode=country_code,
99 vatNumber=vat_number
100 )
101 vat_result = {
102 "valid": result.valid,
103 "name": result.name,
104 "address": result.address
105 }
106 except Exception as e:
107 vat_result = {"error": str(e)}
108 return vat_result
