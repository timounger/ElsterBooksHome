50def convert_json_to_trade_party(trade_party: SellerTradeParty, trade_party_data: dict[str, str | dict[str, str]]) -> None:
51 """!
52 @brief Convert JSON to trade party in drafthorse document
53 @param trade_party : trade party of drafthorse document
54 @param trade_party_data : trade party invoice data
55 """
56 name = trade_party_data["name"] # Unternehmen (BT-27)
57 trade_name = trade_party_data["tradeName"] # Handelsname (BT-28)
58 ident_id = trade_party_data["id"] # Verkäuferkennung (BT-29)
59 trade_id = trade_party_data["tradeId"] # Registernummer (BT-30)
60 ust_id = trade_party_data["vatId"] # Umsatzsteuer-ID (BT-31)
61 tax_id = trade_party_data["taxId"] # Steuernummer (BT-32)
62 legal_notes = trade_party_data.get("legalInfo", "") # Rechtliche Informationen (BT-33)
63 electronic_address = trade_party_data["electronicAddress"] # Elektronische Adresse (BT-34)
64 # Anschrift
65 address_data = trade_party_data["address"]
66 street1 = address_data["line1"] # Straße 1 (BT-35)
67 street2 = address_data["line2"] # Straße 2 (BT-36)
68 postcode = address_data["postCode"] # PLZ (BT-38)
69 city = address_data["city"] # Ort (BT-37)
70 country = address_data["countryCode"] # Land (BT-40) D_COUNTRY_CODE
71 # Kontakt
72 contact_data = trade_party_data["contact"]
73 person = contact_data["name"] # Name (BT-41)
74 email = contact_data["email"] # E-Mail (BT-43)
75 phone = contact_data["phone"] # Telefon (BT-42)
76
77 trade_party.name = name
78 if trade_name:
79 trade_party.legal_organization.trade_name = trade_name
80 if ident_id:
81 trade_party.id = ident_id
82 if trade_id:
83 trade_party.legal_organization.id = trade_id
84 d_tax_data = {TAX_CODE: tax_id, UST_CODE: ust_id}
85 for key, value in d_tax_data.items():
86 if value:
87 tr = TaxRegistration()
88 tr.id._text = value
89 tr.id._scheme_id = key
90 trade_party.tax_registrations.add(tr)
91 if legal_notes:
92 trade_party.description = legal_notes
93 if electronic_address:
94 trade_party.electronic_address.uri_ID = electronic_address
95 if street1:
96 trade_party.address.line_one = street1
97 if street2:
98 trade_party.address.line_two = street2
99 if postcode:
100 trade_party.address.postcode = postcode
101 if city:
102 trade_party.address.city_name = city
103 trade_party.address.country_id = country
104 if person:
105 trade_party.contact.person_name = person
106 if email:
107 trade_party.contact.email.address = email
108 if phone:
109 trade_party.contact.telephone.number = phone

398def eval_factur_xml(xml_file_path: str | bytes, extended: bool = False) -> tuple[bool, str]:
399 """!
400 @brief Evaluate facture xml
401 @param xml_file_path : xml file path or content
402 @param extended : True: check for extended; False: check for EN16931
403 @return valid state
404 """
405 # load XSD scheme
406 warning_text = ""
407 scheme_file = FACTURE_X_EXTENDED_SCHEMA_FILE if extended else FACTURE_X_EN16931_SCHEMA_FILE
408 xsd_file = os.path.join(SCHEMATA_PATH, scheme_file)
409 with open(xsd_file, 'rb') as f:
410 schema_doc = etree.parse(f)
411 schema = etree.XMLSchema(schema_doc)
412
413 xml_doc = None
414 if isinstance(xml_file_path, bytes):
415 xml_doc = etree.fromstring(xml_file_path)
416 else:
417 try:
418 with open(xml_file_path, 'rb') as f:
419 xml_doc = etree.parse(f)
420 except (etree.XMLSchemaParseError, etree.XMLSyntaxError) as _e:
421 xml_doc = None
422
423 if xml_doc is not None:
424 is_valid = schema.validate(xml_doc)
425
426 if not is_valid:
427 warning_text = "Invalid XML Scheme:"
428 for error in schema.error_log: # pylint: disable=not-an-iterable
429 warning_text += f"\n{error.message} (Line {error.line})"
430 log.info(warning_text)
431 else:
432 is_valid = False
433 warning_text = f"Invalid XML File: {xml_file_path}"
434 log.warning(warning_text)
435 return is_valid, warning_text

439 """!

556 gross_amount = (item_quantity * gross_unit_price) / item_basis_quantity
557 vat_amount = gross_amount - net_amount
