132def create_invoice_number(date: QDate, number: Optional[int] = None) -> str | None:
133 """!
134 @brief Get invoice number depend on invoice date.
135 @param date : date to create invoice number
136 @param number : number suffix of invoice number
137 @return invoice number
138 """
139 if number is None:
140 number = 0
141 if number < I_MAX_INVOICE_NUMBER:
142 invoice_number = f"{str((date.year()%100)).zfill(2)}{str(date.month()).zfill(2)}-{str(date.day()).zfill(2)}{str(number).zfill(2)}"
143 else:
144 invoice_number = None
145 return invoice_number

693 def item_price_changed(self, item_index: int, gross_changed: None | bool) -> None:
694 """!
695 @brief Item price changed
696 @param item_index : item index
697 @param gross_changed : changed object True=gross False=net None=other (use last changed)
698 """
699 if not self.lock_auto_price_edit:
700 self.lock_auto_price_edit = True
701 item_dialog: Ui_InvoiceItemData = self.ui_invoice_data.item_widgets[item_index]
702 b_extended = self.cb_extended.isChecked()
703
704 if gross_changed is not None:
705 self.item_gross_changed[item_index] = gross_changed
706
707 net_price = item_dialog.dsb_item_net_unit_price.value()
708 gross_price = item_dialog.dsb_item_gross_unit_price.value()
709 vat_rate = item_dialog.dsb_item_vat_rate.value()
710 item_quantity = item_dialog.dsb_item_quantity.value()
711 item_basis_quantity = item_dialog.dsb_item_basis_quantity.value() if b_extended else 1
712
713 if self.item_gross_changed[item_index]:
714 new_net_price = gross_price / (1 + (vat_rate / 100))
715 net_amount = item_quantity * new_net_price
716 gross_amount = item_quantity * gross_price
717 item_dialog.dsb_item_net_unit_price.setValue(new_net_price)
718 else:
719 new_gross_price = net_price * (1 + (vat_rate / 100))
720 net_amount = item_quantity * net_price
721 gross_amount = item_quantity * new_gross_price
722 item_dialog.dsb_item_gross_unit_price.setValue(new_gross_price)
723 net_amount /= item_basis_quantity
724 gross_amount /= item_basis_quantity
725 tax_amount = gross_amount - net_amount
726 item_dialog.dsb_item_vat_amount.setValue(tax_amount)
727 item_dialog.dsb_item_net_amount.setValue(net_amount)
728 item_dialog.dsb_item_gross_price.setValue(gross_amount)
