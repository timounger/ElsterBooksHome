35 def download_tools(self) -> None:
36 """!
37 @brief Download tools
38 """
39 url = TOOLS_DOWNLOAD_LINK
40 dest_dir = os.path.join(TOOLS_FOLDER, "../")
41 zip_file = os.path.join(dest_dir, "Tools.zip")
42 tools_dir = TOOLS_FOLDER
43 if os.path.isdir(tools_dir):
44 self.status_signal.emit(f"Ordner '{tools_dir}' existiert bereits.")
45 else:
46 # download tools
47 self.status_signal.emit(f"Lade '{url}' herunter...")
48 try:
49 response = requests.get(url, stream=True, timeout=I_TIMEOUT)
50 response.raise_for_status()
51 with open(zip_file, 'wb') as f:
52 for chunk in response.iter_content(chunk_size=8192):
53 f.write(chunk)
54 except requests.exceptions.RequestException as e:
55 self.status_signal.emit(f"Fehler beim Herunterladen der Datei: {e}")
56 return
57
58 # unpack zip
59 self.status_signal.emit(f"Entpacke '{zip_file}' nach '{dest_dir}'...")
60 try:
61 with zipfile.ZipFile(zip_file, 'r') as zip_ref:
62 zip_ref.extractall(dest_dir)
63 except zipfile.BadZipFile as e:
64 self.status_signal.emit(f"Fehler beim Entpacken der ZIP-Datei: {e}")
65 return
66
67 # delete zip
68 self.status_signal.emit(f"Lösche '{zip_file}'...")
69 delete_success = delete_file(zip_file)
70 if not delete_success:
71 self.status_signal.emit(f"Fehler beim Löschen der ZIP-Datei: {e}")
72 self.status_signal.emit("Erfolgreich heruntergeladen")
73

75 """!
76 @brief Run tools download

78 self.download_tools()
79 self.finish_signal.emit()
