165 def create_export(self, e_type: EReportType) -> None:
166 """!
167 @brief Create export
168 @param e_type : report type
169 """
170 i_year = None
171 l_months = None
172 s_period = None
173 match e_type:
174 case EReportType.UST_PRE:
175 s_period = self.ui_export.cb_period.currentText()
176 l_months = self.ui_export.cb_period.currentData()
177 i_year = self.ui_export.sb_year.value()
178 case EReportType.UST | EReportType.EUR | EReportType.GUV | EReportType.DATEV:
179 i_year = self.ui_export.sb_year.value()
180
181 export_report = ExportReport(self.ui, e_type, i_year=i_year, l_months=l_months, period=s_period)
182 self.ui.block_ui()
183 if not os.path.exists(EXPORT_PATH):
184 os.makedirs(EXPORT_PATH)
185 file_name = f"{EXPORT_PATH}/{e_type.value}"
186 if i_year is not None:
187 file_name += f"_{str(i_year)}"
188 if s_period is not None:
189 file_name += f"_{s_period}"
190 file_name += ".xlsx"
191 try:
192 export_report.create_xlsx_report(file_name)
193 except PermissionError:
194 self.ui.unblock_ui()
195 self.ui.set_status(f"Datei kann nicht geschrieben werden: {file_name}", b_highlight=True)
196 else:
197 if e_type != EReportType.DATEV:
198 self.ui.unblock_ui()
199 self.ui.set_status(f"Datei wurde erstellt: {file_name}")
200 with subprocess.Popen(["start", "", file_name], shell=True):
201 pass
202 else:
203 self.ui.unblock_ui()

205 def create_backup(self) -> None:
206 """!
207 @brief Create backup
208 """
209 source_folder = Path(REL_PATH)
210 if source_folder.exists():
211 self.ui.block_ui()
212 now = datetime.now()
213 suffix = now.strftime("%Y-%m-%d_%Hh%Mm%Ss")
214 out_path = f"{EXPORT_PATH}/Backup_{suffix}"
215 shutil.make_archive(out_path, 'zip', source_folder)
216 out_path += ".zip"
217 # copy folder content to clipboard
218 command = f"powershell Set-Clipboard -LiteralPath {out_path}"
219 os.system(command)
220 open_explorer(out_path)
221 self.ui.unblock_ui()
222 self.ui.set_status(f"Sicherung wurde erstellt: {out_path}.zip")
223 else:
224 self.ui.set_status(f"Der Ordner existiert nicht: {source_folder}", b_highlight=True)

226 def combine_pdf(self) -> None:
227 """!
228 @brief Combine PDF
229 """
230 l_select_files, _ = QFileDialog.getOpenFileNames(parent=self.ui, caption="PDF kombinieren",
231 directory=self.ui.model.get_last_path(),
232 filter=PDF_FILE_TYPES)
233 if l_select_files:
234 now = datetime.now()
235 suffix = now.strftime("%Y-%m-%d_%Hh%Mm%Ss")
236 _path, file_name = os.path.split(l_select_files[0])
237 file_name, _file_type = os.path.splitext(file_name)
238 out_file = os.path.join(EXPORT_PATH, f"{file_name}_Combined_{suffix}.pdf")
239 combined_pdf = fitz.open()
240 for pdf_file in l_select_files:
241 actual_pdf = fitz.open(pdf_file)
242 combined_pdf.insert_pdf(actual_pdf)
243 actual_pdf.close()
244 combined_pdf.save(out_file)
245 combined_pdf.close()
246 open_explorer(out_file)
