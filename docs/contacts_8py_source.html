107def add_contact(path: str, add: bool, contact: dict[EContactFields, str],

109 """!
110 @brief Add or actualize contact.

117 s_id = set_general_json_data(contact, CONTACT_TYPE, EContactFields.JSON_TYPE,

122 id_field = EContactFields.ID if (contact_id is not None) else None
