52 def set_table_data(self, update: bool = False, rename: bool = False, update_dashboard: bool = True) -> None:
53 """!
54 @brief Read document data and update table.
55 @param update : update status of JSON file
56 @param rename : rename status of file name
57 @param update_dashboard : update dashboard
58 """
59 clear_dialog_data(self)
60 self.l_data = read_document(self.ui.model.data_path)
61 l_data = []
62 for document in self.l_data:
63 if update:
64 add_document(self.ui.model.data_path, self.ui.model.git_add, document, document[EDocumentFields.ID], rename=rename)
65 l_entry = []
66 l_entry.append(CellData(text=document[EDocumentFields.DOCUMENT_DATE], right_align=True, is_date=True))
67 l_entry.append(CellData(icon=self.ui_document.get_attach_icon(document[EDocumentFields.ATTACHMENT])))
68 l_entry.append(CellData(text=document[EDocumentFields.DESCRIPTION]))
69 l_entry.append(CellData(text=document[EDocumentFields.ID]))
70 l_data.append(l_entry)
71 self.ui_document.update_table(l_data)
72 self.ui_document.table.setColumnHidden(len(L_ROW_DESCRIPTION) - 1, True)

99 if found_doc is not None:
