61 def set_table_data(self, update: bool = False, rename: bool = False, update_dashboard: bool = True) -> None:
62 """!
63 @brief Read income data and update table.
64 @param update : update status of JSON file
65 @param rename : rename status of file name
66 @param update_dashboard : update dashboard
67 """
68 clear_dialog_data(self)
69 self.l_data = read_income(self.ui.model.data_path)
70 l_data = []
71 for income in self.l_data:
72 if update:
73 export_income(self.ui.model.data_path, self.ui.model.git_add, income, income[EReceiptFields.ID], rename=rename)
74 status, icon = get_status(income, self.ui.tab_settings.company_data[COMPANY_DEFAULT_FIELD][ECompanyFields.PAYMENT_DAYS])
75 l_entry = []
76 l_entry.append(CellData(text=status, icon=icon))
77 l_entry.append(CellData(text=income[EReceiptFields.PAYMENT_DATE], right_align=True, is_date=True))
78 l_entry.append(CellData(text=income[EReceiptFields.INVOICE_DATE], right_align=True, is_date=True))
79 l_entry.append(CellData(text=income[EReceiptFields.INVOICE_NUMBER]))
80 l_entry.append(CellData(text=income[EReceiptFields.DELIVER_DATE]))
81 l_entry.append(CellData(text=income[EReceiptFields.GROUP]))

83 l_entry.append(CellData(text=income[EReceiptFields.TRADE_PARTNER]))
84 l_entry.append(CellData(text=income[EReceiptFields.DESCRIPTION]))
85 l_entry.append(CellData(text=income[EReceiptFields.COMMENT]))
86 if isinstance(income[EReceiptFields.AMOUNT_NET], (float, int)):
87 l_entry.append(CellData(text=f"{income[EReceiptFields.AMOUNT_NET]:.2f} EUR", right_align=True))
88 self.total_net += income[EReceiptFields.AMOUNT_NET]

90 l_entry.append(CellData(text="", right_align=True))

92 l_entry.append(CellData(text=f"{income[EReceiptFields.AMOUNT_GROSS]:.2f} EUR", right_align=True))
93 self.total_gross += income[EReceiptFields.AMOUNT_GROSS]

121 def on_item_double_clicked(self, row: int, col: int, _value: str) -> None:
122 """!
123 @brief Callback for double click on table entry.
124 @param row : clicked row index
125 @param col : clicked column index
126 @param _value : value of clicked cell
127 """
128 model = self.ui_income.table.model()
129 uid_index = model.index(row, len(L_RECEIPT_ROW_DESCRIPTION) - 1)
130 uid = model.data(uid_index, Qt.ItemDataRole.DisplayRole)
131
132 if uid is not None:
133 if col == I_ATTACH_IDX:
134 self.open_invoice_file(uid)

136 found_income = next((income for income in self.l_data if income[EReceiptFields.ID] == uid), None)
137 if found_income is not None:
138 ReceiptDialog(self.ui, found_income, uid, receipt_type=EReceiptType.INCOME)
139 self.set_table_data()
140 else:
