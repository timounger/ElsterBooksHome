34def parse_number(s_string: str) -> float | int | None:
35 """!
36 @brief parse number to int or float
37 @param s_string : string to check
38 @return number
39 """
40 try:
41 number = float(s_string)
42 if number.is_integer():
43 number = int(number)

195 def vat_id_changed(self, vat_id: str) -> None:
196 """!
197 @brief VAT ID changed
198 @param vat_id : VAT id
199 """
200 self.vat_validator.terminate()
201 if vat_id:
202 if check_vat_format(vat_id):
203 self.lbl_vat_status.setText("Überprüfung läuft...")
204 self.lbl_vat_status.setStyleSheet("color: grey")
205 self.vat_validator.vat_id = vat_id
206 self.vat_validator.start()
207 else:
208 self.lbl_vat_status.setText("VAT Nummer ungültiges Format")
209 self.lbl_vat_status.setStyleSheet("color: grey")
210 else:
211 self.lbl_vat_status.setText("")
