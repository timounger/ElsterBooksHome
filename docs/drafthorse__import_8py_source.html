116def set_spin_box_read_only(widget: QSpinBox | QDoubleSpinBox, value: int | float, max_decimals: Optional[int] = None) -> None:
117 """!
118 @brief Set spin box as read only without focus
119 @param widget : widget
120 @param value : value to set
121 @param max_decimals : reduce decimals to minimum required with this maximum
122 """
123 if max_decimals is not None:
124 if isinstance(widget, QDoubleSpinBox):
125 d = Decimal(str(value)).normalize()
126 if d == d.to_integral():
127 decimals = 0
128 else:
129 decimals = -int(d.as_tuple().exponent)
130 decimals = min(decimals, max_decimals)
131 widget.setDecimals(decimals)
132 else:
133 raise TypeError("Widget is not QDoubleSpinBox to set Decimals")
134 widget.setValue(value)
135 widget.setReadOnly(True)
136 widget.setButtonSymbols(QAbstractSpinBox.ButtonSymbols.NoButtons)
137 widget.setFocusPolicy(Qt.FocusPolicy.NoFocus) # disable mouse focus
138 widget.setAttribute(Qt.WidgetAttribute.WA_TransparentForMouseEvents, True) # disable mouse focus

164def set_optional_entry(widget: QLineEdit | QPlainTextEdit | QSpinBox | QDoubleSpinBox | QDateEdit,
165 value: str,
166 other_widget: Optional[QLabel | list[QLabel]] = None) -> None:

168 @brief Set optional entry

171 @param other_widget : other widgets
172 """

177 set_spin_box_read_only(widget, value)
178 elif isinstance(widget, QLineEdit):
179 set_line_edit_read_only(widget, value)
180 elif isinstance(widget, QPlainTextEdit):
181 set_plain_text_read_only(widget, value)
182 else:
183 raise TypeError(f"Invalid Optional widget '{type(widget)}'")
184 else:
185 widget.hide()
186 if other_widget is not None:
187 if isinstance(other_widget, list):
188 for other_widget_item in other_widget:
189 other_widget_item.hide()
190 else:
191 other_widget.hide()

198def extract_xml_from_pdf(pdf_path: str) -> bytes | None:
199 """!
200 @brief Extract XML content from PDF
201 @param pdf_path : PDF file path
202 @return XML content
203 """
204 file_content = None
205 doc = fitz.open(pdf_path)
206
207 # check attachments
208 for i in range(doc.embfile_count()):
209 attachment = doc.embfile_info(i)
210 file_name = attachment['name']
211 if file_name.endswith(XML_TYPE):
212 file_content = doc.embfile_get(attachment['name'])
213 break
214
215 # in some cases: check for embedded x-invoice
216 if file_content is None:
217 for i in range(doc.xref_length()):
218 try:
219 obj = doc.xref_object(i, compressed=True)
220 if "/Type/EmbeddedFile" in obj:
221 # filename = doc.xref_get_key(i, "UF")[1] or f"embedded_{i}.xml" # get filename
222 file_content = doc.xref_stream(i) # extract stream
223 break
224 except Exception:
225 file_content = None
226 return file_content

276def import_invoice(xml_content: Optional[bytes], is_income: bool, income_group: str = "", expenditure_group: str = "") -> dict[EReceiptFields, Any]:
277 """!
278 @brief Import invoice
279 @param xml_content : XML content
280 @param is_income : True: is income; False: is expenditure
281 @param income_group : income group
282 @param expenditure_group : expenditure group
283 @return data
284 """
285 if xml_content is not None:
286 doc = Document.parse(xml_content)
287 invoice_date = doc.header.issue_date_time._value
288 if invoice_date:
289 invoice_date = datetime.strptime(str(invoice_date), DATE_FORMAT_XML)
290 invoice_date = invoice_date.strftime(DATE_FORMAT_JSON)
291 else:
292 invoice_date = ""
293 if is_income:
294 trade_partner = str(doc.trade.agreement.buyer.name)
295 else:
296 trade_partner = str(doc.trade.agreement.seller.name)
297 invoice_number = str(doc.header.id)
298
299 deliver_date = doc.trade.delivery.event.occurrence._value
300 if deliver_date:
301 deliver_date = datetime.strptime(str(deliver_date), DATE_FORMAT_XML)
302 deliver_date = deliver_date.strftime(DATE_FORMAT_JSON)
303 else:
304 deliver_date = ""
305
306 description = ""
307 for child in doc.trade.items.children:
308 description = str(child.product.name)[:100] # limit description
309 break
310
311 gross = doc.trade.settlement.monetary_summation.grand_total._amount
312 net = doc.trade.settlement.monetary_summation.tax_basis_total._amount
313
314 b_success = True
315 else:
316 b_success = False
317
318 receipt: dict[EReceiptFields, Any] = {}
319 if b_success:
320 receipt[EReceiptFields.TRADE_PARTNER] = trade_partner
321 receipt[EReceiptFields.DESCRIPTION] = description
322 receipt[EReceiptFields.INVOICE_DATE] = invoice_date
323 receipt[EReceiptFields.INVOICE_NUMBER] = invoice_number
324 receipt[EReceiptFields.DELIVER_DATE] = deliver_date
325 receipt[EReceiptFields.AMOUNT_GROSS] = float(gross)
326 receipt[EReceiptFields.AMOUNT_NET] = float(net)
327 receipt[EReceiptFields.BAR] = False
328 receipt[EReceiptFields.GROUP] = income_group if is_income else expenditure_group
329 receipt = fill_data(D_RECEIPT_TEMPLATE, receipt)
330 return receipt
