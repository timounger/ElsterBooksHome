14def normalize_decimal(value: Any, max_decimals: int = 4) -> Decimal:
15 """!
16 @brief Normalize decimal
17 @param value : value
18 @param max_decimals : maximal decimals
19 @return decimal value
20 """
21 d = Decimal(str(value)).quantize(Decimal("1." + "0" * max_decimals), rounding=ROUND_HALF_UP)
22 s = format(d.normalize(), 'f') # Convert to f-format to avoid exponents
23 if '.' in s: # If there is a decimal point, remove only superfluous decimal places.
