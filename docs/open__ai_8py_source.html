79 def ask_openai(self, text: str) -> InvoiceData | None:
80 """!
81 @brief Ask OpenAI
82 @param text : document text
83 @return invoice data
84 """
85 document_data = None
86 messages = [
87 {
88 "role": "user",
89 "content": create_user_message(text),
90 }]
91
92 try:
93 response = self.openai_client.beta.chat.completions.parse(
94 model=self.model, # only this model support structured output
95 messages=messages,
96 response_format=InvoiceData,
97 )
98 document_data = response.choices[0].message.parsed
99 except Exception as e:
100 log.warning("Error during ask OpenAI: %s", e)
101 document_data = None
102 else:
103 document_data = validate_answer(document_data)
104
105 return document_data
