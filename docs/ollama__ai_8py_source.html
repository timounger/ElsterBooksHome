154 """!
155 @brief Set auto model for Ollama AI

157 for model, req_ram in D_OLLAMA_RAM_RES.items():
158 if (req_ram + I_RAM_OFFSET <= self.i_ram_size) and (model in self.l_models):

163 def ask_ollama(self, text: str) -> InvoiceData | None:
164 """!
165 @brief Ask ollama
166 @param text : document text
167 @return invoice data
168 """
169 document_data = None
170 messages = [
171 {
172 "role": "user",
173 "content": create_user_message(text),
174 }]
175
176 try:
177 response = chat(messages=messages, model=self.model, format=InvoiceData.model_json_schema())
178 document_data = InvoiceData.model_validate_json(response.message.content) # pylint: disable=no-member
179 except Exception as e:
180 log.warning("Error during ask Ollama: %s", e)
181 document_data = None
182 else:
183 document_data = validate_answer(document_data)
184
185 return document_data
