135 """!
136 @brief Get invoice number depend on invoice date.
137 @param date : date to create invoice number
138 @param number : number suffix of invoice number
139 @return invoice number
140 """
141 invoice_number = None
142 if False: # TODO
143 # 1. Fortlaufende Nummern (einfachste Form): 1, 2 ,3 ...
144 # 2. Nummer mit Jahresangabe: YYYY-0001
145 # 3. Nummer mit Monatsangabe: YYYYMM-0001
146 # 4. Kombination aus Datum und laufender Nummer : YYYYMMDD-001
147
148 # Placeholder
149 # {YYYY} 4-stelliges Jahr
150 # {YY} 2-stelliges Jahr
151 # {MM} Monat (01–12)
152 # {DD} Tag (01–31)
153 # {SEQ} Fortlaufende Nummer
154 # {CUST} Kunden-ID
155 # {PROJ} Projekt-ID
156
157 # reset option for {SEQ}
158 # none Nie zurücksetzen, fortlaufend
159 # daily Täglich zurücksetzen
160 # monthly Monatlich zurücksetzen
161 # yearly Jährlich zurücksetzen
162
163 # führende nullen
164 # e.g. use: {SEQ:daily}
165 # e.g. use: {SEQ:3}
166 # e.g. {SEQ:<reset>:<length>}
167
168 template = "{YY}{MM}-{SEQ:daily:2}"
169 last_sequences = {}
170
171 date_replacements = {
172 "{YYYY}": str(date.year()),
173 "{YY}": str((date.year() % 100)).zfill(2),
174 "{MM}": str(date.month()).zfill(2),
175 "{DD}": str(date.day()).zfill(2),
176 }
177
178 for key, val in date_replacements.items():
179 template = template.replace(key, val)
180
181 # Sequenz-Pattern erkennen
182 seq_pattern = r"\{SEQ(?::([^}:]*))?(?::(\d+))?\}"
183 match = re.search(seq_pattern, template)
184
185 if match:
186 reset_mode = match.group(1) or "none"
187 seq_length = int(match.group(2)) if match.group(2) else 0
188
189 # Schlüssel für last_sequences bestimmen
190 if reset_mode == "daily":
191 seq_key = f"{date.year():04d}-{date.month():02d}-{date.day():02d}"
192 elif reset_mode == "monthly":
193 seq_key = f"{date.year():04d}-{date.month():02d}"
194 elif reset_mode == "yearly":
195 seq_key = f"{date.year():04d}"
196 elif reset_mode == "none":
197 seq_key = "global"
198 else:
199 raise ValueError(f"Reset-Modus '{reset_mode}' nicht unterstützt.")
200
201 # Nächste Sequenz bestimmen
202 seq = last_sequences.get(seq_key, 0) + 1
203 last_sequences[seq_key] = seq
204
205 # Sequenz formatieren
206 formatted_seq = str(seq).zfill(seq_length) if seq_length > 0 else str(seq)
207
208 # Platzhalter ersetzen
209 template = re.sub(seq_pattern, formatted_seq, template)
210
211 invoice_number = template
212 else:
