82 def exception_hook(self, exc_type: type[BaseException], exc_value: BaseException, exc_traceback: Optional[TracebackType] = None) -> None:
83 """!
84 @brief Custom exception hook. It is triggered each time an uncaught exception occurs.
85 @param exc_type : exception type
86 @param exc_value : exception value
87 @param exc_traceback : exception traceback
88 """
89 if issubclass(exc_type, KeyboardInterrupt):
90 # ignore keyboard interrupt to support console application
91 sys.__excepthook__(exc_type, exc_value, exc_traceback)
92 else:
93 log_msg = "\n".join([f"{exc_type.__name__}: {exc_value}",
94 "".join(traceback.format_tb(exc_traceback))])
95 log.error(log_msg)
96
97 # trigger message box show
98 self.exception_caught.emit(log_msg)
