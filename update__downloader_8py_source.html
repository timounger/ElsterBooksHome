62 def download_update(self) -> bool:
63 """!
64 @brief Download update
65 @return success_status : download success status
66 """
67 success_status = False
68 url = f"https://github.com/{__owner__}/{__repo__}/releases/download/v{self.latest_version}/{__title__}.exe"
69 # download update
70 try:
71 response = requests.get(url, stream=True, timeout=I_TIMEOUT)
72 response.raise_for_status()
73
74 # Gesamtlänge aus dem Header (falls vorhanden)
75 total_length = response.headers.get('content-length')
76 if total_length is not None:
77 total_length = int(total_length)
78 else:
79 total_length = 0 # unbekannt
80
81 downloaded = 0
82 last_status_text = ""
83 with open(NEW_VERSION_FILE, 'wb') as f:
84 for chunk in response.iter_content(chunk_size=8192):
85 f.write(chunk)
86 downloaded += len(chunk)
87 if total_length > 0:
88 percent = downloaded / total_length * 100
89 mb_done = downloaded / (1024 * 1024)
90 mb_total = total_length / (1024 * 1024)
91 status_text = f"{mb_done:.1f} von {mb_total:.1f} MB ({percent:.1f}%)"
92 else:
93 mb_done = downloaded / (1024 * 1024)
94 status_text = f"{mb_done:.1f} MB heruntergeladen"
95 if status_text != last_status_text:
96 self.status_signal.emit(status_text)
97 last_status_text = status_text
98 except requests.exceptions.RequestException:
99 self.status_signal.emit("Update fehlgeschlagen")
100 else:
101 success_status = True
102 return success_status
103
104 def run(self) -> None:
105 """!
106 @brief Run tools download
107 """
108 success_status = self.download_update()
109 self.finish_signal.emit(success_status)
