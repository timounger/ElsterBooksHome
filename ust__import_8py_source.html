18def check_ust(pdf_path: str) -> bool:
19 """!
20 @brief Check if document is pre tax
21 @param pdf_path : PDF file path
22 @return status if document is pre tax
23 """
24 b_is_pre_tax = False
25 with fitz.open(pdf_path) as pdf_document:
26 for page_num in range(pdf_document.page_count):
27 page = pdf_document[page_num]
28 text = page.get_text()

35def import_ust(pdf_path: str, is_income: bool) -> dict[EReceiptFields, Any]:
36 """!
37 @brief Read items from PDF file
38 @param pdf_path : PDF file path
39 @param is_income : True: is income; False: is expenditure
40 @return data
41 """
42 b_neg_amount = False
43 event = EReceiptGroup.UST.value
44 invoice_date = ""
45 amount: float | int = 0
46 b_pre_amount = False
47 b_pre_year = False
48 tax_office = "Unbekannt"
49 with fitz.open(pdf_path) as pdf_document:
50 for page_num in range(pdf_document.page_count):
51 page = pdf_document[page_num]
52 text = page.get_text()
53 lines = text.split('\n')
54 for _i, line in enumerate(lines):
55 # print(line)
56 if "Erstellungsdatum" in line:
57 invoice_date = line.split()[1]
58 elif "Finanzamt:" in line:
59 tax_office = line.split()[1]
60 elif b_pre_amount:
61 s_amount = line.split()[0]
62 amount = convert_amount(s_amount)
63 if amount < 0:
64 b_neg_amount = True
65 amount = abs(amount)
66 b_pre_amount = False
67 elif b_pre_year:
68 event += f" {line}"
69 b_pre_year = False
70 elif line == "Minuszeichen voranstellen -":
71 b_pre_amount = True
72 elif line == "Kalenderjahr":
73 b_pre_year = True
74
75 receipt: dict[EReceiptFields, Any] = {}
76 b_success = bool(is_income == b_neg_amount)
77 if b_success:
78 receipt[EReceiptFields.TRADE_PARTNER] = f"FK {tax_office}"
79 receipt[EReceiptFields.DESCRIPTION] = event
80 receipt[EReceiptFields.INVOICE_DATE] = invoice_date
81 receipt[EReceiptFields.AMOUNT_GROSS] = amount
82 receipt[EReceiptFields.AMOUNT_NET] = amount
83 receipt[EReceiptFields.BAR] = False
84 receipt[EReceiptFields.GROUP] = EReceiptGroup.UST
85 receipt = fill_data(D_RECEIPT_TEMPLATE, receipt)
86 return receipt
