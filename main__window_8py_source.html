57 def eventFilter(self, watched: Optional[QObject], event: Optional[QEvent]) -> bool: # pylint: disable=invalid-name
58 """!
59 @brief Filter to prevent tip event (statusbar message no longer disappears on menu hover).
60 @param watched : object
61 @param event : arrived event
62 @return return event filter
63 """
64 if isinstance(event, QStatusTipEvent):
65 b_return = True
66 else:
67 b_return = super().eventFilter(watched, event)
68 return b_return

251 btn.setText("Ja")

255 btn.setText("Nein")
256 box = QCheckBox("Nicht mehr anzeigen")
257 dialog.setCheckBox(box)
258 dialog.close()
259 choice = dialog.exec()

261 b_update = choice not in [QMessageBox.StandardButton.No, QMessageBox.StandardButton.Close, QMessageBox.StandardButton.Cancel]
262
263 if is_checked:
264 write_update_version(newer_tool_version) # write newest version for don't remember again
265 if b_update:

278 def set_status(self, text: str, b_warning: bool = False, i_timeout: Optional[int] = None, b_highlight: bool = False) -> None:
279 """!
280 @brief Logs a status message to status bar (with timer) and logging handler
281 @param text : text to set
282 @param b_warning : [True] Text is a warning; [False] normal info
283 @param b_highlight : [True] highlight text; [False] normal text
284 @param i_timeout : timeout for statustext in "ms". If None use default time
285 """
286 if b_warning:
287 log.warning(text)
288 else:
289 log.info(text)
290
291 if self.gui_locked:
292 self.statusbar.showMessage("Das Fenster ist gesperrt!")
293 self.statusbar.setStyleSheet(f"background-color: {LOCKED_STYLE};")
294 else:
295 if not (not b_warning and self.b_warning_active):
296 if hasattr(self, "status_timer"):
297 if i_timeout is None:
298 if b_warning:
299 i_timeout = STATUS_WARNING_TEXT_TIME
300 else:
301 i_timeout = STATUS_HIGHLIGHT_TEXT_TIME if b_highlight else STATUS_TEXT_TIME
302 if i_timeout is None:
303 self.status_timer.stop()
304 else:
305 self.status_timer.start(i_timeout)
306 if b_warning:
307 foreground = None
308 background = WARNING_STYLE if self.model.c_monitor.is_light_theme() else WARNING_STYLE_DARK
309 self.b_warning_active = True
310 else:
311 foreground = HIGHLIGHT_STYLE if b_highlight else None
312 background = DEFAULT_STYLE
313 self.b_warning_active = False
314 self.statusbar.showMessage(text)
315 if background is not None:
316 self.statusbar.setStyleSheet(f"background-color: {background};")
317 if foreground is not None:
318 self.statusbar.setStyleSheet(f"color: {foreground};")
319
320 def clear_status(self, b_override: bool = False) -> None:
321 """!
322 @brief Logs a status message to status bar (with timer) and logging handler
323 @param b_override : status if actual status should override
324 """
325 if not self.gui_locked:
326 if not self.status_timer.isActive() or (b_override and not self.b_warning_active):
327 s_user_text = ""
328 self.statusbar.showMessage(s_user_text)
329 self.statusbar.setStyleSheet(f"background-color: {DEFAULT_STYLE};")
330 self.b_warning_active = False
