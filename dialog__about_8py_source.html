120 def update_btn_clicked(self) -> None:
121 """!
122 @brief Update button clicked
123 """
124 if not self.b_download_finished:
125 if self.timer.isActive():
126 self.timer.stop()
127 # buttons
128 self.btn_update.hide()
129 # icon
130 container_size = 48 # slightly larger than your icon, space for rotation
131 icon_size = 32 # Original size of the SVG
132 b_light_theme = self.ui.model.c_monitor.is_light_theme()
133 renderer = QSvgRenderer(ICON_UPDATE_LIGHT if b_light_theme else ICON_UPDATE_DARK)
134 pixmap = QPixmap(container_size, container_size)
135 pixmap.fill(Qt.GlobalColor.transparent)
136 painter = QPainter(pixmap)
137 offset = (container_size - icon_size) / 2
138 renderer.render(painter, QRectF(offset, offset, icon_size, icon_size)) # Draw SVG centered
139 painter.end()
140 self.lbl_update_icon.show()
141 self.lbl_update_icon.setPixmap(pixmap)
142 self.lbl_update_icon.setFixedSize(container_size, container_size)
143 self.lbl_update_icon.setAlignment(Qt.AlignmentFlag.AlignCenter)
144 self.lbl_update_icon._original_pixmap = pixmap # Save original for rotation
145 # label
146 self.lbl_update_status.show()
147 lbl_text = "Update wird heruntergeladen"
148 self.lbl_update_status.setText(lbl_text)
149
150 self.start_label_rotation()
151 self.update_downloader.start()
152 else:
153 generate_and_start_updater_script()
154 sys.exit()

156 def rotate(self) -> None:
157 """!
158 @brief Rotate update icon
159 """
160 original = self.lbl_update_icon._original_pixmap
161 size = original.width()
162 rotated_pixmap = QPixmap(size, size)
163 rotated_pixmap.fill(Qt.GlobalColor.transparent)
164
165 painter = QPainter(rotated_pixmap)
166 painter.setRenderHint(QPainter.RenderHint.SmoothPixmapTransform)
167
168 transform = QTransform()
169 transform.translate(size / 2, size / 2)
170 transform.rotate(self.angle)
171 transform.translate(-size / 2, -size / 2)
172
173 painter.setTransform(transform)
174 painter.drawPixmap(0, 0, original)
175 painter.end()
176
177 step_deg = 5
178 self.lbl_update_icon.setPixmap(rotated_pixmap)
179 self.angle = (self.angle + step_deg) % 360
