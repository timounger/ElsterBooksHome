59 def set_table_data(self, update: bool = False, rename: bool = False, update_dashboard: bool = True) -> None:
60 """!
61 @brief Read expenditure data and update table.
62 @param update : update status of JSON file
63 @param rename : rename status of file name
64 @param update_dashboard : update dashboard
65 """
66 clear_dialog_data(self)
67 self.l_data = read_expenditure(self.ui.model.data_path)
68 l_data = []
69 for expenditure in self.l_data:
70 if update:
71 export_expenditure(self.ui.model.data_path, self.ui.model.git_add, expenditure, expenditure[EReceiptFields.ID], rename=rename)
72 status, icon = get_status(expenditure, self.ui.tab_settings.company_data[COMPANY_DEFAULT_FIELD][ECompanyFields.PAYMENT_DAYS])
73 l_entry = []
74 l_entry.append(CellData(text=status, icon=icon))
75 l_entry.append(CellData(text=expenditure[EReceiptFields.PAYMENT_DATE], is_date=True))
76 l_entry.append(CellData(text=expenditure[EReceiptFields.INVOICE_DATE], is_date=True))
77 l_entry.append(CellData(text=expenditure[EReceiptFields.INVOICE_NUMBER]))
78 l_entry.append(CellData(text=expenditure[EReceiptFields.DELIVER_DATE]))
79 l_entry.append(CellData(text=expenditure[EReceiptFields.GROUP]))
80 l_entry.append(CellData(icon=self.ui_expenditure.get_attach_icon(expenditure[EReceiptFields.ATTACHMENT])))
81 l_entry.append(CellData(text=expenditure[EReceiptFields.TRADE_PARTNER]))
82 l_entry.append(CellData(text=expenditure[EReceiptFields.DESCRIPTION]))
83 l_entry.append(CellData(text=expenditure[EReceiptFields.COMMENT]))
84 if isinstance(expenditure[EReceiptFields.AMOUNT_NET], (float, int)):
85 l_entry.append(CellData(text=f"{expenditure[EReceiptFields.AMOUNT_NET]:.2f} EUR", right_align=True))
86 self.total_net += expenditure[EReceiptFields.AMOUNT_NET]
87 else:
88 l_entry.append(CellData(text="", right_align=True))
89 if isinstance(expenditure[EReceiptFields.AMOUNT_GROSS], (float, int)):
90 l_entry.append(CellData(text=f"{expenditure[EReceiptFields.AMOUNT_GROSS]:.2f} EUR", right_align=True))
91 self.total_gross += expenditure[EReceiptFields.AMOUNT_GROSS]
92 self.l_value.append(expenditure[EReceiptFields.AMOUNT_GROSS])
93 self.l_invoice_date.append(expenditure[EReceiptFields.INVOICE_DATE])
94 self.l_payment_date.append(expenditure[EReceiptFields.PAYMENT_DATE])
95 else:
96 l_entry.append(CellData(text="", right_align=True))
97 l_entry.append(CellData(text=expenditure[EReceiptFields.ID]))
98 l_data.append(l_entry)
99 self.ui_expenditure.update_table(l_data)
100 self.ui_expenditure.table.setColumnHidden(len(L_RECEIPT_ROW_DESCRIPTION) - 1, True)
101 if update_dashboard:
102 self.ui.tab_dashboard.update_dashboard_data()

119 def on_item_double_clicked(self, row: int, col: int, _value: str) -> None:
120 """!
121 @brief Callback for double click on table entry.
122 @param row : clicked row index
123 @param col : clicked column index
124 @param _value : value of clicked cell
125 """
126 model = self.ui_expenditure.table.model()
127 uid_index = model.index(row, len(L_RECEIPT_ROW_DESCRIPTION) - 1)
128 uid = model.data(uid_index, Qt.ItemDataRole.DisplayRole)
129
130 if uid is not None:
131 if col == I_ATTACH_IDX:
132 self.open_invoice_file(uid)
133 else:
134 found_expenditure = next((expenditure for expenditure in self.l_data if expenditure[EReceiptFields.ID] == uid), None)
135 if found_expenditure is not None:
136 ReceiptDialog(self.ui, found_expenditure, uid, receipt_type=EReceiptType.EXPENDITURE)
137 self.set_table_data()
138 else:
139 self.ui.set_status("Expenditure UID not found", True) # state not possible
