270 """
271 class ContextActions(str, Enum):
272 """!
273 @brief Possible context menu actions
274 """
275 ACTION_CREATE_INVOICE = "Rechnung erstellen"
276 ACTION_DELETE_ENTRY = "Eintrag löschen"
277
278 index = self.table.indexAt(point)
279 if index.isValid():
280 b_light_theme = self.ui.model.c_monitor.is_light_theme()
281 menu = QMenu(self.ui)
282 if self.b_create_invoice:
283 icon = QIcon(ICON_INVOICE_LIGHT if b_light_theme else ICON_INVOICE_DARK)
284 menu.addAction(icon, ContextActions.ACTION_CREATE_INVOICE.value)
285 icon = QIcon(ICON_DELETE_LIGHT if b_light_theme else ICON_DELETE_DARK)
286 menu.addAction(icon, ContextActions.ACTION_DELETE_ENTRY.value)
287 selected_action = menu.exec(self.table.viewport().mapToGlobal(point))
288 if selected_action:
289 row = index.row()
290 model = self.table.model()
291 uid_index = model.index(row, len(self.l_table_header) - 1)
292 uid = model.data(uid_index, Qt.ItemDataRole.DisplayRole)
293 match selected_action.text():
294 case ContextActions.ACTION_CREATE_INVOICE:
295 self.ui.tab_income.create_invoice(uid)
296 case ContextActions.ACTION_DELETE_ENTRY:
297 if (self.delete_fnc is not None) and (self.update_table_func is not None):
298 self.delete_fnc(self.ui.model.data_path, uid)
299 self.update_table_func()
300 self.ui.set_status("Eintrag gelöscht")
301 case _:
302 log.warning("Unknown header context menu action selected: %s", selected_action.text())
