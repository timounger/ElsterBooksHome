176 def get_relevant_data_status(self, data: dict[EReceiptFields, str], sheet_type: EReportSheet) -> bool:
177 """!
178 @brief Get relevant data status
179 @param data : data
180 @param sheet_type : sheet type
181 @return status if data relevant
182 """
183 if sheet_type == EReportSheet.PRE_TAX:
184 b_data_relevant = bool(data[EReceiptFields.DESCRIPTION].startswith(f"{EReportType.UST_PRE.value} {self.i_year}"))
185 else:
186 date_string = data[self.date_field]
187 if date_string and is_date_format(date_string): # check date format for invalid data
188 data_date = datetime.strptime(date_string, DATE_FORMAT_JSON)
189 b_ust_report = bool(self.e_type in [EReportType.UST_PRE, EReportType.UST])
190 b_data_is_ust = bool(data[EReceiptFields.GROUP] in [EReceiptGroup.UST_VA, EReceiptGroup.UST])
191 b_hide_data = bool(b_ust_report and b_data_is_ust)
192 if self.i_year is not None:
193 if b_data_is_ust:
194 if data_date.month == 1 and data_date.day <= UST_REGULATION_DAYS:
195 relevant_year = data_date.year - 1 # relevant for year before
196 else:
197 relevant_year = data_date.year
198 else:
199 relevant_year = data_date.year
200 b_year_relevant = bool(relevant_year == self.i_year)
201 else:
202 b_year_relevant = True
203 if self.l_months is not None:
204 b_month_relevant = bool(data_date.month in self.l_months)
205 else:
206 b_month_relevant = True
207 b_data_relevant = (not b_hide_data) and b_year_relevant and b_month_relevant
208 else:
209 b_data_relevant = False
210 return b_data_relevant

212 def create_receipt_sheet(self, xls_creator: XLSCreator, sheet_type: EReportSheet, l_receipt_data: list[dict[EReceiptFields, Any]],
213 l_invert_data: Optional[list[dict[EReceiptFields, Any]]] = None, tab_color: Optional[str] = None) -> int:
214 """!
215 @brief Create receipt worksheet
216 @param xls_creator : XLS creator
217 @param sheet_type : sheet type
218 @param l_receipt_data : receipt data
219 @param l_invert_data : invert data for pre tax
220 @param tab_color : tab color
221 @return line number of sum
222 """
223 sheet_title = sheet_type.value
224 worksheet = xls_creator.workbook.create_sheet(sheet_title)
225 worksheet.title = sheet_title
226 if tab_color is not None:
227 worksheet.sheet_properties.tabColor = tab_color
228 Worksheet.set_printer_settings(worksheet, paper_size=9, orientation="landscape")
229 xls_creator.set_page_marcins(worksheet, left=1.5, right=1.5, top=1.0, bottom=1.0)
230 worksheet.column_dimensions["A"].width = 10 # INVOICE_DATE
231 worksheet.column_dimensions["B"].width = 10 # INVOICE_NUMBER
232 worksheet.column_dimensions["C"].width = 20 # TRADE_PARTNER
233 worksheet.column_dimensions["D"].width = 20 # DESCRIPTION
234 worksheet.column_dimensions["E"].width = 10 # PAYMENT_DATE
235 worksheet.column_dimensions["F"].width = 4 # BAR
236 worksheet.column_dimensions["G"].width = 10 # GROUP
237 worksheet.column_dimensions["H"].width = 17 # COMMENT
238 worksheet.column_dimensions["I"].width = 17 # DELIVER_DATE
239 worksheet.column_dimensions["J"].width = 14 # AMOUNT_GROSS
240 worksheet.column_dimensions["K"].width = 14 # AMOUNT_NET
241 worksheet.column_dimensions["L"].width = 14 # USt (calculated)
242 worksheet.column_dimensions["M"].width = 6 # % USt (calculated)
243 worksheet.column_dimensions["N"].width = 36 # ID
244
245 if self.e_type != EReportType.EXPORT_TOTAL:
246 worksheet.column_dimensions["H"].hidden = True # COMMENT
247 worksheet.column_dimensions["I"].hidden = True # DELIVER_DATE
248 worksheet.column_dimensions["N"].hidden = True # ID
249
250 l_datev = []
251 l_datev.append(["Datum", "Belegnummer", "Handelspartner", "Beschreibung", "Bezahlt", "Kategorie", "Netto", "Brutto"])
252
253 xls_creator.set_cell(worksheet, 1, 1, sheet_title, bold=True, font_size=28)
254
255 l_data = copy.deepcopy(l_receipt_data)
256 if l_invert_data is not None: # only for pretax not None
257 for invert_data in copy.deepcopy(l_invert_data):
258 invert_data[EReceiptFields.AMOUNT_GROSS] = -invert_data[EReceiptFields.AMOUNT_GROSS]
259 invert_data[EReceiptFields.AMOUNT_NET] = -invert_data[EReceiptFields.AMOUNT_NET]
260 l_data.append(invert_data)
261
262 # Get relevant data
263 d_vat_rate_breakdown: dict[float | int, float | int] = {}
264 for tax_rate in self.tax_rates:
265 d_vat_rate_breakdown[tax_rate] = 0
266 l_relevant_data = []
267 for entry in l_data:
268 if self.get_relevant_data_status(entry, sheet_type):
269 l_relevant_data.append(entry)
270 # Vat Rate breakdown
271 vat_rate = calc_vat_rate(entry[EReceiptFields.AMOUNT_GROSS], entry[EReceiptFields.AMOUNT_NET])
272 d_vat_rate_breakdown[vat_rate] = 0 # create empty vat to know present rates for sort
273 d_vat_rate_breakdown_sorted = dict(sorted(d_vat_rate_breakdown.items(), key=lambda item: item[0], reverse=True))
274
275 receipt_idx = 0
276 i_row = receipt_idx
277 for entry in l_relevant_data:
278 i_row = receipt_idx + 3
279 xls_creator.set_cell(worksheet, i_row, 1, entry[EReceiptFields.INVOICE_DATE], number_format=NUMBER_FORMAT_DATETIME)
280 xls_creator.set_cell(worksheet, i_row, 2, entry[EReceiptFields.INVOICE_NUMBER])
281 xls_creator.set_cell(worksheet, i_row, 3, entry[EReceiptFields.TRADE_PARTNER])
282 xls_creator.set_cell(worksheet, i_row, 4, entry[EReceiptFields.DESCRIPTION])
283 xls_creator.set_cell(worksheet, i_row, 5, entry[EReceiptFields.PAYMENT_DATE], number_format=NUMBER_FORMAT_DATETIME)
284 if entry[EReceiptFields.BAR]:
285 xls_creator.set_cell(worksheet, i_row, 6, "X")
286 xls_creator.set_cell(worksheet, i_row, 7, entry[EReceiptFields.GROUP])
287 xls_creator.set_cell(worksheet, i_row, 8, entry[EReceiptFields.COMMENT])
288 xls_creator.set_cell(worksheet, i_row, 9, entry[EReceiptFields.DELIVER_DATE])
289 xls_creator.set_cell(worksheet, i_row, 10, entry[EReceiptFields.AMOUNT_GROSS], number_format=NUMBER_FORMAT_EUR)
290 xls_creator.set_cell(worksheet, i_row, 11, entry[EReceiptFields.AMOUNT_NET], number_format=NUMBER_FORMAT_EUR)
291 xls_creator.set_cell(worksheet, i_row, 12, f"=(J{i_row}-K{i_row})", number_format=NUMBER_FORMAT_EUR)
292 xls_creator.set_cell(worksheet, i_row, 13, f"=(IF(K{i_row}=0, 0, (J{i_row}/K{i_row})-1))", number_format=NUMBER_FORMAT_PERCENT)
293 xls_creator.set_cell(worksheet, i_row, 14, entry[EReceiptFields.ID])
294 # Vat Rate breakdown
295 if sheet_type in [EReportSheet.INCOME, EReportSheet.EXPENDITURE]:
296 vat_rate = calc_vat_rate(entry[EReceiptFields.AMOUNT_GROSS], entry[EReceiptFields.AMOUNT_NET])
297 tax_amount = entry[EReceiptFields.AMOUNT_GROSS] - entry[EReceiptFields.AMOUNT_NET]
298 d_vat_rate_breakdown_sorted[vat_rate] = d_vat_rate_breakdown_sorted.get(vat_rate, 0) + entry[EReceiptFields.AMOUNT_NET]
299 vat_column = 15 + (2 * list(d_vat_rate_breakdown_sorted.keys()).index(vat_rate))
300 xls_creator.set_cell(worksheet, i_row, vat_column, entry[EReceiptFields.AMOUNT_NET], number_format=NUMBER_FORMAT_EUR)
301 xls_creator.set_cell(worksheet, i_row, vat_column + 1, tax_amount, number_format=NUMBER_FORMAT_EUR)
302 receipt_idx += 1
303 # add group data for eur
304 if self.e_type in [EReportType.EUR, EReportType.GUV, EReportType.EXPORT_TOTAL]:
305 group = entry[EReceiptFields.GROUP]
306 data_date = datetime.strptime(entry[self.date_field], DATE_FORMAT_JSON)
307 month = data_date.month if (data_date.year == self.i_year) else 12
308 b_data_is_ust = bool(entry[EReceiptFields.GROUP] in [EReceiptGroup.UST_VA, EReceiptGroup.UST])
309 year = data_date.year - 1 if (b_data_is_ust and (data_date.month == 1) and (data_date.day <= UST_REGULATION_DAYS)) else data_date.year
310 match sheet_type:
311 case EReportSheet.INCOME:
312 if self.e_type in [EReportType.EUR, EReportType.GUV]:
313 self.d_income_groups[group] = self.d_income_groups.get(group, 0) + entry[EReceiptFields.AMOUNT_NET]
314 self.d_income_month[month] = self.d_income_month.get(month, 0) + entry[EReceiptFields.AMOUNT_GROSS]
315 if self.e_type == EReportType.EXPORT_TOTAL:
316 self.d_income_year[year] = self.d_income_year.get(year, 0) + entry[EReceiptFields.AMOUNT_GROSS]
317 case EReportSheet.EXPENDITURE:
318 if self.e_type in [EReportType.EUR, EReportType.GUV]:
319 self.d_expenditure_groups[group] = self.d_expenditure_groups.get(group, 0) + entry[EReceiptFields.AMOUNT_NET]
320 self.d_expenditure_month[month] = self.d_expenditure_month.get(month, 0) + entry[EReceiptFields.AMOUNT_GROSS]
321 if self.e_type == EReportType.EXPORT_TOTAL:
322 self.d_expenditure_year[year] = self.d_expenditure_year.get(year, 0) + entry[EReceiptFields.AMOUNT_GROSS]
323 # set data for CSV export
324 if self.e_type == EReportType.DATEV:
325 l_append_data = []
326 l_append_data.append(entry[EReceiptFields.INVOICE_DATE]) # Datum
327 l_append_data.append(entry[EReceiptFields.INVOICE_NUMBER]) # Belegnummer
328 l_append_data.append(entry[EReceiptFields.TRADE_PARTNER]) # Handelspartner
329 l_append_data.append(entry[EReceiptFields.DESCRIPTION]) # Beschreibung
330 l_append_data.append(entry[EReceiptFields.PAYMENT_DATE]) # Bezahlt
331 # use only first word of group if is number
332 group = entry[EReceiptFields.GROUP]
333 words = group.split()
334 if len(words) > 0:
335 if words[0].isdigit():
336 group = words[0]
337 l_append_data.append(group) # Kategorie
338 l_append_data.append(str(entry[EReceiptFields.AMOUNT_NET]).replace(".", ",")) # Netto
339 l_append_data.append(str(entry[EReceiptFields.AMOUNT_GROSS]).replace(".", ",")) # Brutto
340 l_datev.append(l_append_data)
341
342 # set title after iterate over entry for known column because vat rate breakdown
343 header = ["Datum", "Re-Nr.", "Handelspartner", "Beschreibung", "Bezahlt", "Bar", "Gruppe", "Kommentar", "Lieferzeitraum", "Brutto", "Netto", "USt.", "% USt.", "ID"]
344 if sheet_type in [EReportSheet.INCOME, EReportSheet.EXPENDITURE]:
345 group_letter_start = get_column_letter(len(header) + 1)
346 for vat_rate in d_vat_rate_breakdown_sorted:
347 header_row = len(header) + 1
348 header_letter = get_column_letter(header_row)
349 header_row_next = header_row + 1
350 header_letter_next = get_column_letter(header_row_next)
351 worksheet.column_dimensions[header_letter].width = 20
352 worksheet.column_dimensions[header_letter_next].width = 20
353 worksheet.merge_cells(f"{header_letter}1:{header_letter_next}1")
354 xls_creator.set_cell(worksheet, 1, header_row, f"Steuersatz {vat_rate}%", bold=True, italic=True, align="center", align_vert="bottom")
355 header += [f"Grundbetrag zu {vat_rate}%", f"Steuerbetrag zu {vat_rate}%"]
356 if sheet_type == EReportSheet.INCOME:
357 self.d_income_vat_rate_sum_cells[vat_rate] = (header_row, i_row + 1)
358 worksheet.column_dimensions.group(group_letter_start, header_letter_next, hidden=True)
359 for i, s_title in enumerate(header):
360 xls_creator.set_cell(worksheet, 2, i + 1, s_title, bold=True)
361
362 if self.e_type == EReportType.DATEV:
363 datev_file_name = None
364 export_dir = os.path.dirname(self.file_name)
365 match sheet_type:
366 case EReportSheet.INCOME:
367 datev_file_name = f"{export_dir}/Rechnungseingangsbelege.csv"
368 case EReportSheet.EXPENDITURE:
369 datev_file_name = f"{export_dir}/Rechnungsausgangsbelege.csv"
370 if datev_file_name is not None:
371 with open(datev_file_name, mode="w", encoding="utf-8-sig", newline="") as file: # utf-8-sig for umlauts
372 writer = csv.writer(file, delimiter=";")
373 for datev_entry in l_datev:
374 writer.writerow(datev_entry)
375 if receipt_idx == 0: # set correct row if no data
376 receipt_idx = 1
377 i_row = receipt_idx + 2
378 xls_creator.set_table(worksheet, max_col=len(header), max_row=receipt_idx + 2, min_col=1, min_row=2)
379
380 i_row += 1
381 sum_header = [""] * len(header)
382 sum_header[0] = "Summe"
383 sum_header[9] = f"=SUM({sheet_title}[Brutto])"
384 sum_header[10] = f"=SUM({sheet_title}[Netto])"
385 sum_header[11] = f"=SUM({sheet_title}[USt.])"
386 if len(sum_header) > 14:
387 for i, vat_rate in enumerate(d_vat_rate_breakdown_sorted):
388 sum_header[14 + (i * 2)] = f"=SUM({sheet_title}[Grundbetrag zu {vat_rate}%])"
389 sum_header[14 + (i * 2) + 1] = f"=SUM({sheet_title}[Steuerbetrag zu {vat_rate}%])"
390 for i, s_title in enumerate(sum_header):
391 xls_creator.set_cell(worksheet, i_row, i + 1, s_title, bold=True, number_format=NUMBER_FORMAT_EUR, fill_color=COLOR_YELLOW)
392
393 # xls_creator.workbook.active = workbook[sheet_title]
394
395 return i_row

397 def add_tax_to_overview_sheet(self, xls_creator: XLSCreator) -> None:
398 """!
399 @brief Add tax data to overview sheet
400 @param xls_creator : XLS creator
401 """
402 worksheet = self.sheet_overview
403 i_row = 9
404 match self.e_type:
405 case EReportType.UST_PRE | EReportType.UST:
406 sum_row = 5
407 sum_row_letter = get_column_letter(sum_row)
408 xls_creator.set_cell(worksheet, i_row, 1, "Art der Besteuerung:")
409 b_agreed_cost = self.ui.tab_settings.company_data[COMPANY_BOOKING_FIELD][ECompanyFields.AGREED_COST]
410 tax_method = "Soll-Versteuerung (bei Rechnungsstellung)" if b_agreed_cost else "Ist-Versteuerung (bei Eingang der Zahlung)"
411 xls_creator.set_cell(worksheet, i_row, 2, tax_method)
412 i_row += 2
413 # income
414 xls_creator.set_cell(worksheet, i_row, 1, "Steuerpflichtige Umsätze")
415 xls_creator.set_cell(worksheet, i_row, sum_row - 2, "Bemessungsgrundlage")
416 xls_creator.set_cell(worksheet, i_row, sum_row, "Steuer")
417 tax_sum_row_start = i_row + 1
418 for vat_rate, cell_position in self.d_income_vat_rate_sum_cells.items():
419 column_number, row_number = cell_position
420 i_row += 1
421 tax_sum_row_end = i_row
422 net_value_cell = f"{get_column_letter(column_number)}{row_number}"
423 tax_value_cell = f"{get_column_letter(column_number+1)}{row_number}"
424 xls_creator.set_cell(worksheet, i_row, 1, f"zum Steuersatz von {vat_rate} Prozent")
425 xls_creator.set_cell(worksheet, i_row, sum_row - 1, f"={EReportSheet.INCOME.value}!{net_value_cell}", number_format=NUMBER_FORMAT_EUR)
