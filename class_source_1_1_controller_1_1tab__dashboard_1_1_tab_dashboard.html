150 def update_chart(self) -> None:
151 """!
152 @brief Update chart.
153 """
154 # data
155 l_income = calc_chart_data(self.ui.tab_income.l_invoice_date, self.ui.tab_income.l_value)
156 l_expenditure = calc_chart_data(self.ui.tab_expenditure.l_invoice_date, self.ui.tab_expenditure.l_value)
157 l_expenditure = [-x for x in l_expenditure] # expenditure are negative
158 d_data = {
159 "Einnahmen": l_income,
160 "Ausgaben": l_expenditure
161 }
162
163 today = datetime.today()
164 month_names = []
165 for i in range(0, I_MONTH_IN_YEAR):
166 month = today.month - i
167 year = today.year
168 if month <= 0:
169 month += I_MONTH_IN_YEAR
170 year -= 1
171 month_names.append(L_MONTH_NAMES_SHORT[month - 1])
172 month_names.reverse()
173
174 l_numbers = [value for values_list in d_data.values() for value in values_list]
175
176 min_value = min(l_numbers)
177 max_value = max(l_numbers)
178 max_range = max(abs(min_value), abs(max_value))
179 max_range = round_to_nearest_integer(max_range)
180 steps = 5
181
182 series = QBarSeries()
183 series.setBarWidth(0.8)
184 series.setLabelsVisible(True)
185 series.setLabelsPosition(QBarSeries.LabelsPosition.LabelsOutsideEnd)
186 for key, value in d_data.items():
187 bar_set = QBarSet(key)
188 income_color = QColor("lightgreen") if self.ui.model.c_monitor.is_light_theme() else QColor("green")
189 expenditure_color = QColor(255, 71, 77) if self.ui.model.c_monitor.is_light_theme() else QColor("red")
190 q_color = income_color if key == "Einnahmen" else expenditure_color
191 bar_set.setBrush(q_color)
192 bar_set.append(value)
193 bar_set.setLabelColor(QColor("black"))
194 bar_set.setLabelFont(QFont("Arial", 8, weight=QFont.Weight.Bold))
195 series.append(bar_set)
196
197 # Create chart
198 chart = QChart()
199 chart.addSeries(series)
200 # chart.setTitle("Umsatz")
201 chart.setAnimationOptions(QChart.AnimationOption.SeriesAnimations) # SeriesAnimations GridAxisAnimations AllAnimations
202 chart.setAnimationDuration(1000)
203
204 # x-axis
205 axis_x = QBarCategoryAxis()
206 axis_x.append(month_names)
207 # axis_x.setTitleText("Zeitraum")
208 chart.addAxis(axis_x, Qt.AlignmentFlag.AlignBottom)
209 series.attachAxis(axis_x)
210
211 # y-axis
212 axis_y = QValueAxis()
213 axis_y.setLabelFormat("%d") # hide decimal digits
214 axis_y.setTickCount(steps)
215 axis_y.setRange(-max_range, max_range)
216 axis_y.setTitleText("Betrag [€]")
217 chart.addAxis(axis_y, Qt.AlignmentFlag.AlignLeft)
218 series.attachAxis(axis_y)
219
220 # Create chart view
221 color = "white" if self.ui.model.c_monitor.is_light_theme() else "lightgrey" # default dark QColor(33, 33, 33)
222 chart.setBackgroundBrush(QColor(color))
223 chart_view = QChartView(chart)
224 chart_view.setRenderHint(QPainter.RenderHint.Antialiasing) # Antialiasing TextAntialiasing SmoothPixmapTransform
225
226 # delete all old QChartView(s) before set new to prevent lagging
227 for i in reversed(range(self.ui_dashboard.gridLayout_2.count())):
228 item = self.ui_dashboard.gridLayout_2.itemAt(i)
229 widget = item.widget()
230 if isinstance(widget, QChartView):
231 self.ui_dashboard.gridLayout_2.takeAt(i)
232 widget.deleteLater()
233
234 # Set central widget
235 self.ui_dashboard.gridLayout_2.addWidget(chart_view, 0, 0, 1, self.ui_dashboard.gridLayout_2.columnCount())
