31def check_write_access(path: str) -> bool:
32 """!
33 @brief Check write access of path
34 @param path : path to check
35 @return write access status
36 """
37 try: # test write access
38 s_test_file = path + "/" + S_WRITE_TEST_FILE_NAME
39 with open(s_test_file, mode="w", encoding="utf-8") as file:
40 file.write("Write Access Test")
41 os.remove(s_test_file)
42 except BaseException:
43 b_access = False
44 else:
45 b_access = True
46 return b_access
