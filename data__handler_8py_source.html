206def calc_vat_rate(gross: float, net: float) -> int | float:
207 """!
208 @brief Calculate vat rate to string and try to convert to integer string
209 @param gross : gross value
210 @param net : net value
211 @return vat rate
212 """
213 if net != 0:
214 vat_rate = round(((gross / net) - 1) * 100, 4)
215 vat_rate_int = int(round(vat_rate, 0))
216 gross_reference = net * (1 + (vat_rate_int / 100))
217 diff = abs(gross - gross_reference)
218 if diff < 0.02: # allow one cent deviation (less than 2 cent for deviation)
219 select_vat_rate = vat_rate_int
220 else:
221 select_vat_rate = vat_rate
222 else:
223 select_vat_rate = 0
224 return select_vat_rate

227def convert_amount(string: str) -> int | float:
228 """!
229 @brief Convert amount
230 @param string : string to convert
231 @return converted string
232 """
233 string_value = string.replace(" ", "").replace(",", ".")
234 last_point_index = string_value.rfind('.')
235 if last_point_index != -1:
236 string_value = string_value[:last_point_index].replace('.', '') + string_value[last_point_index:]
237 float_value = float(string_value)
238 amount: int | float = float_value
239 if float_value % 1 == 0:
240 amount = int(float_value)
241 return amount

277def fill_data(template: dict[Any, Any], data: dict[Any, Any]) -> dict[Any, Any]:
278 """!
279 @brief Fill data to template dictionary.
280 @param template : default template dictionary
281 @param data : actual data dictionary to fill in template dictionary
282 @return filled data
283 """
284 filled_data = {}
285 for key, default_value in template.items():
286 if isinstance(default_value, dict):
287 if isinstance(data.get(key), dict):
288 filled_data[key] = fill_data(default_value, data[key])
289 else:
290 filled_data[key] = fill_data(default_value, {})
291 else:
292 filled_data[key] = data.get(key, default_value)
293 return filled_data

296def get_file_name(title_data: str | list[str], uid: Optional[str] = None) -> str:
297 """!
298 @brief Get file name from title.
299 connect single title data, delete invalid file name characters and add short UID
300 @param title_data : title
301 @param uid : UID
302 @return valid file name
303 """
304 s_title = ""
305 if not isinstance(title_data, list):
306 title_data = [title_data]
307 for i, data in enumerate(title_data):
308 if data != "":
309 if i != 0:
310 s_title += "_"
311 s_title += data.replace("\n", " ").strip() # write in single row
312 s_title = re.sub(r'[^\w\s.-]', '', s_title)
313 s_title = s_title[:200] # max file title length
314 if uid is not None:
315 s_title += f"_{uid[:8]}"
316 return s_title

319def get_file_name_content(file_path: str) -> tuple[str | None, str | None]:
320 """!
321 @brief Get file name content
322 @param file_path : file path
323 @return file date and content
324 """
325 file_date = None
326 file_content = None
327 file_name = os.path.basename(file_path)
328 if re.match(r"^\d{4}_\d{2}_\d{2}_", file_name):
329 file_date = datetime.strptime(file_name[:10], "%Y_%m_%d").strftime("%d.%m.%Y")
330 file_name_without_ending = os.path.splitext(file_name)[0]
331 if file_date is None:
332 file_content = file_name_without_ending
333 else:
334 file_content = file_name_without_ending[11:] # remove prefix yyyy_mm_dd_
335 return file_date, file_content

513def find_file(path: str, uid: str, file_name: Optional[str] = None, file_type: Optional[str] = None) -> str | None:
514 """!
515 @brief Find file from title or UID.
516 @param path : find file in this path
517 @param uid : UID
518 @param file_name : possible attachment file name
519 @param file_type : file type [None] find every file type
520 @return founded file
521 """
522 found_file = None
523 if os.path.exists(path):
524 if file_name is not None:
525 file_path = os.path.join(path, file_name)
526 name, _file_type = os.path.splitext(file_name)
527 if os.path.isfile(file_path) and name.endswith(uid[:8]):
528 if (file_type is None) or file_name.endswith(file_type):
529 found_file = file_path
530 if found_file is None:
531 for actual_file in os.listdir(path):
532 if (file_type is None) or actual_file.endswith(file_type):
533 name, _file_type = os.path.splitext(actual_file)
534 if name.endswith(uid[:8]):
535 found_file = os.path.join(path, actual_file)
536 break
537 return found_file

577def add_json(add: bool, data: dict[str, Any], title: str, uid: str, json_export_path: str, id_field: Optional[str] = None, rename: bool = False) -> None:
578 """!
579 @brief Add or actualize JSON file
580 @param add : GIT add status
581 @param data : data to add or actualize
582 @param title : title name
583 @param uid : UUID to create short id for suffix name or find file
584 @param json_export_path: add JSON to this file path
585 @param id_field : field of ID [None] JSON is new
586 @param rename : status if existing file should renamed (JSON and appendix)
587 """
588 file_path = os.path.join(json_export_path, f"{title}{JSON_TYPE}")
589 if (id_field is None) and not rename:
590 write_json_file(file_path, data)
591 if add:
592 git_add(file_path)
593 else:
594 if id_field is None:
595 log.warning("No field to add JSON")
596 return
597 found_file_path = find_json(json_export_path, uid, id_field)
598 if found_file_path is not None:
599 old_data = read_json_file(found_file_path)
600 if old_data[id_field] != uid:
601 raise ValueError("Invalid UID to update")
602 write_json_file(found_file_path, data)
603 if rename:
604 if found_file_path != file_path:
605 git_rename(found_file_path, file_path)

670def add_receipt(d_receipt_data: dict[EReceiptFields, Any], receipt_type: str, json_export_path: str, appendix_export_path: str, add: bool,
671 uid: Optional[str] = None, appendix_file: Optional[str] = None, rename: bool = False, number_in_title: bool = False) -> None:
672 """!
673 @brief Add or actualize receipt (data and appendix) to accounting.
674 @param d_receipt_data : receipt data
675 @param receipt_type : type parameter to write in JSON file
676 @param json_export_path : add JSON to this file path
677 @param appendix_export_path : add appendix to this file path
678 @param add : GIT add status
679 @param uid : UUID of receipt [None] receipt is new to add
680 @param appendix_file : appendix file to add with JSON data
681 @param rename : status if existing file should renamed (JSON and appendix)
682 @param number_in_title : status if invoice number is in file title

693 if (appendix_file is not None) or rename:
694 if appendix_file is None:
695 if d_receipt_data[EReceiptFields.ATTACHMENT]:
696 _, file_extension = os.path.splitext(d_receipt_data[EReceiptFields.ATTACHMENT])
697 file_extension = file_extension.lower()
698 else:
699 file_extension = PDF_TYPE # only for compatible with old version without existing attachment
700 else:
701 _, file_extension = os.path.splitext(appendix_file)
702 file_extension = file_extension.lower()
703 d_receipt_data[EReceiptFields.ATTACHMENT] = f"{title}{file_extension}"
704 id_field = EReceiptFields.ID if (uid is not None) else None
705 instance = fill_data(D_RECEIPT_TEMPLATE, d_receipt_data)
706 add_json(add, instance, title, s_id, json_export_path, id_field=id_field, rename=rename)
707 if (appendix_file is not None) or rename:
708 add_appendix(title, s_id, appendix_export_path, add, appendix_file=appendix_file)

724 """!

749 actual_file_path = os.path.join(attachment_path, actual_file)
750 git_delete(actual_file_path)

768def get_git_repo(file_path: str) -> bool:
769 """!
770 @brief Get git Repo with configuration.
771 @param file_path : file to add
772 @return status if git repo exist
773 """

775 if B_USE_GIT_CMD:

784 test_repo = git.Repo(os.path.abspath(REL_PATH), search_parent_directories=True)

786 except git.exc.GitError:

807def git_rename(old_file_path: str, new_file_path: str) -> None:
808 """!
809 @brief Rename file on GIT.
810 @param old_file_path : old file name
811 @param new_file_path : new file name
812 """
813 if old_file_path == new_file_path:
814 log.warning("File name same to rename (%s)", old_file_path)
815 else:
816 old_file_path = os.path.abspath(old_file_path)
817 new_file_path = os.path.abspath(new_file_path)
818 repo = get_git_repo(old_file_path)
819 if repo:
820 if B_USE_GIT_CMD:
821 _result = git_cmd(["-C", os.path.abspath(REL_PATH), "mv", old_file_path, new_file_path])
822 else:
823 try:
824 repo.index.move([old_file_path, new_file_path])
825 except git.GitCommandError as err:
826 log.error("Git Rename failed (%s): %s", new_file_path, err)
827 else:
828 p_old_file_path = Path(old_file_path)
829 p_new_file_path = Path(new_file_path)
830 p_old_file_path.rename(p_new_file_path)

870 @brief Check for Git Changes.
871 @return changes status
872 """
873 b_changes = False
874 s_changes = ""

877 if B_USE_GIT_CMD:

881 else:
882 s_changes = "" # TODO

884 if repo.is_dirty(untracked_files=False):
885 log.debug("Changes found to commit")
886 for item in repo.index.diff(None):
887 log.debug("Changed: %s", item.a_path)
888 for item in repo.untracked_files:
889 log.debug("Untracked: %s", item)
890 b_changes = True
891 else:
892 log.debug("No changes to commit")
893 except git.exc.InvalidGitRepositoryError:
894 log.debug("Path is no Git Path")

965def get_status(invoice: dict[EReceiptFields, Any], payment_days: int) -> tuple[str, str]:
966 """!
967 @brief Get invoice status.
968 @param invoice : invoice data
969 @param payment_days : payment days
970 @return status value and icon as tuple
971 """
972 if not invoice[EReceiptFields.INVOICE_DATE]:
973 icon = ICON_CIRCLE_WHITE
974 status = EStatus.DESIGN
975 elif invoice[EReceiptFields.PAYMENT_DATE]:
976 icon = ICON_CIRCLE_GREEN
977 status = EStatus.PAID
978 elif invoice[EReceiptFields.AMOUNT_GROSS] == 0:
979 icon = ICON_CIRCLE_GREEN
980 status = EStatus.NO_PRICE
981 else:
982 invoice_date = datetime.strptime(invoice[EReceiptFields.INVOICE_DATE], DATE_FORMAT_JSON)
983 current_datetime = datetime.now()
984 diff_datetime = current_datetime - invoice_date
985 if diff_datetime.days > payment_days:
986 icon = ICON_CIRCLE_RED
987 status = EStatus.DUE
988 else:
989 icon = ICON_CIRCLE_ORANGE
990 status = EStatus.PENDING
991 return status.value, icon
